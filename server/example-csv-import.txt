# Exemplo de CSV para importar dados de passageiros

# Formato básico (campos obrigatórios):
firstName,lastName,email,flightNumber,departureCity,arrivalCity,departureDate

# Exemplo 1 - Passageiro simples:
João,Silva,joao@email.com,BR123,São Paulo,Rio de Janeiro,2024-02-01

# Exemplo 2 - Com campos opcionais:
firstName,lastName,email,phone,passportNumber,nationality,dateOfBirth,seatNumber,flightNumber,departureCity,arrivalCity,departureDate,ticketClass,price,status
Maria,Santos,maria@email.com,+55 21 88888-2222,BR789012,brasileira,1990-07-22,15B,LA1234,São Paulo,Los Angeles,2024-01-15,business,4500.00,confirmed

# Exemplo 3 - Múltiplos passageiros:
firstName,lastName,email,flightNumber,departureCity,arrivalCity,departureDate,ticketClass,price
Carlos,Oliveira,carlos@email.com,BR456,São Paulo,Brasília,2024-02-15,economy,1200.00
Ana,Costa,ana@email.com,BR789,Rio de Janeiro,Salvador,2024-02-20,business,2800.00
Pedro,Ferreira,pedro@email.com,BR012,Brasília,Manaus,2024-02-25,first,3500.00

# Notas importantes:
# - Sempre inclua o cabeçalho (primeira linha)
# - Campos obrigatórios: firstName, lastName, email, flightNumber, departureCity, arrivalCity, departureDate
# - Campos opcionais: phone, passportNumber, nationality, dateOfBirth, seatNumber, ticketClass, price, status
# - Use vírgulas para separar campos
# - Não use aspas a menos que seja necessário
# - O campo status padrão é "confirmed" se não especificado
